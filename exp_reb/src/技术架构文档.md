# 服务部署与路由实验框架 技术架构文档

## 1. 概述

本框架用于实现和评测微服务系统的部署算法和路由算法，重点关注**Baseline算法**的实现和评测，以及后续**Our算法**的集成与对比。

框架参考以下论文的建模方法：
- **论文一** (IEEE TPDS 2023): 微服务部署与路由的排队网络建模
- **论文二** (IEEE TSC 2024): 双时间尺度优化框架
- **论文三** (GSTC): M/M/C排队模型 + 服务-服务器亲和度建模
- **Our算法**: 基于globecom.pdf的创新算法（待实现）

```python
# core/topology/base.py

class Node:
    """计算节点"""
    def __init__(self, node_id: str, cpu_cores: int):
        self.node_id = node_id
        self.cpu_cores = cpu_cores              # CPU核心数
        self.used_cpu = 0                       # 已使用CPU
        self.services = {}                       # {service_id: instance_count}
        self.position = (0.0, 0.0, 0.0)         # 地理位置(可选)

    @property
    def available_cpu(self) -> int:
        """可用CPU核心数"""
        return self.cpu_cores - self.used_cpu

    def can_deploy(self, cpu_needed: int) -> bool:
        """检查是否可以部署"""
        return self.available_cpu >= cpu_needed

    def deploy_service(self, service_id: str, instances: int, cpu_per_instance: float):
        """部署服务实例"""
        self.services[service_id] = instances
        self.used_cpu += int(instances * cpu_per_instance)

    def remove_service(self, service_id: str):
        """移除服务"""
        if service_id in self.services:
            self.used_cpu -= int(self.services[service_id] * self._cpu_per_instance)
            del self.services[service_id]


class Link:
    """网络链路"""
    def __init__(self, src: str, dst: str, bandwidth: float, delay: float):
        self.src = src
        self.dst = dst
        self.bandwidth = bandwidth    # 带宽 (Mbps)
        self.delay = delay            # 传播延迟 (ms)


class Topology:
    """网络拓扑"""
    def __init__(self):
        self.nodes: Dict[str, Node] = {}
        self.links: List[Link] = []
        self.adjacency: Dict[str, List[str]] = {}  # 邻接表

    def add_node(self, node: Node):
        """添加节点"""
        self.nodes[node.node_id] = node
        self.adjacency[node.node_id] = []

    def add_link(self, link: Link):
        """添加链路"""
        self.links.append(link)
        self.adjacency[link.src].append(link.dst)

    def get_shortest_path(self, src: str, dst: str) -> List[str]:
        """
        Dijkstra最短路径算法
        Returns: [node_id_1, node_id_2, ..., node_id_k]
        """
        import heapq
        distances = {node_id: float('inf') for node_id in self.nodes}
        distances[src] = 0
        previous = {node_id: None for node_id in self.nodes}
        pq = [(0, src)]

        while pq:
            current_dist, current = heapq.heappop(pq)
            if current_dist > distances[current]:
                continue
            if current == dst:
                break
            for neighbor in self.adjacency.get(current, []):
                link = self._get_link(current, neighbor)
                if link:
                    distance = current_dist + link.delay
                    if distance < distances[neighbor]:
                        distances[neighbor] = distance
                        previous[neighbor] = current
                        heapq.heappush(pq, (distance, neighbor))

        # 重建路径
        path = []
        current = dst
        while current:
            path.append(current)
            current = previous[current]
        path.reverse()
        return path if path[0] == src else []

    def get_communication_delay(self, src: str, dst: str) -> float:
        """获取两节点间通信延迟"""
        path = self.get_shortest_path(src, dst)
        if len(path) <= 1:
            return 0.0
        total_delay = 0.0
        for i in range(len(path) - 1):
            link = self._get_link(path[i], path[i + 1])
            if link:
                total_delay += link.delay
        return total_delay

    def _get_link(self, src: str, dst: str) -> Optional[Link]:
        """获取两点间的链路"""
        for link in self.links:
            if link.src == src and link.dst == dst:
                return link
        return None

    def get_nodes_by_resource(self, cpu_needed: int) -> List[Node]:
        """筛选满足CPU需求的节点"""
        return [n for n in self.nodes.values() if n.can_deploy(cpu_needed)]
```

# core/service/microservice.py

@dataclass
class MicroService:
    """微服务定义"""
    service_id: str
    cpu_per_instance: float      # 每个实例CPU核心数
    processing_rate: float        # 处理速率 μ (请求/秒)
    data_size: float             # 输出数据大小 (MB)

    def calc_service_time(self) -> float:
        """计算平均服务时间 = 1/μ"""
        return 1.0 / self.processing_rate


@dataclass
class ServiceChain:
    """服务链/请求类型"""
    chain_id: str
    services: List[str]           # 服务ID列表 [s1, s2, ..., sk]
    arrival_rate: float           # 到达率 λ (请求/秒)
    max_latency: float            # 最大容忍延迟 (ms)

    def __post_init__(self):
        self.service_count = len(self.services)


@dataclass
class ServiceInstance:
    """服务实例（运行时）"""
    service_id: str
    node_id: str
    instance_id: int
    queue_length: int = 0
    arrival_rate: float = 0.0     # 累计到达率

    @property
    def utilization(self) -> float:
        """利用率"""
        if self.processing_rate <= 0:
            return 0.0
        return self.arrival_rate / self.processing_rate


@dataclass
class DeploymentPlan:
    """部署方案"""
    # {(service_id, node_id): instance_count}
    placement: Dict[Tuple[str, str], int] = field(default_factory=dict)

    def add_instance(self, service_id: str, node_id: str, count: int = 1):
        key = (service_id, node_id)
        self.placement[key] = self.placement.get(key, 0) + count

    def get_instances(self, service_id: str) -> int:
        """获取服务总实例数"""
        return sum(v for (s, n), v in self.placement.items() if s == service_id)

    def get_node_instances(self, node_id: str) -> Dict[str, int]:
        """获取节点上部署的服务"""
        return {s: v for (s, n), v in self.placement.items() if n == node_id}

    def get_node_cpu_usage(self, node_id: str, service_configs: Dict[str, MicroService]) -> int:
        """计算节点CPU使用量"""
        total_cpu = 0
        for (s, n), v in self.placement.items():
            if n == node_id and s in service_configs:
                total_cpu += int(v * service_configs[s].cpu_per_instance)
        return total_cpu

### 3.3 排队模型模块 (core/queueing)

```python
# core/queueing/mmc.py

class MMCQueue:
    """
    M/M/C 排队模型 (无限容量)

    公式:
    - 服务强度: ρ = λ / (c * μ)
    - 等待概率 (Erlang-C): Pw
    - 平均队列长度: Lq
    - 平均等待时间: Wq = Lq / λ
    - 平均响应时间: W = Wq + 1/μ
    """

    def __init__(self, arrival_rate: float, service_rate: float,
                 num_servers: int):
        """
        Args:
            arrival_rate: 到达率 λ
            service_rate: 服务率 μ (每个服务器)
            num_servers: 服务器数量 c
        """
        self.arrival_rate = arrival_rate
        self.service_rate = service_rate
        self.num_servers = num_servers

    @property
    def rho(self) -> float:
        """服务强度"""
        return self.arrival_rate / (self.num_servers * self.service_rate)

    def is_stable(self) -> bool:
        """系统是否稳定 (ρ < 1)"""
        return self.rho < 1.0

    def calc_prob_wait(self) -> float:
        """
        计算等待概率 (Erlang-C公式)
        Pw = ( (c*ρ)^c / (c!*(1-ρ)) ) * (Σ_{k=0}^{c-1} (c*ρ)^k/k! + (c*ρ)^c/(c!*(1-ρ)) )^-1
        """
        if self.rho >= 1.0:
            return 1.0

        c = self.num_servers
        rho = self.rho

        # 计算分子 (c*ρ)^c / c!
        numerator = ((c * rho) ** c) / self._factorial(c)

        # 计算分母部分
        sum_k = sum((c * rho) ** k / self._factorial(k) for k in range(c))
        denominator = sum_k + ((c * rho) ** c) / (self._factorial(c) * (1 - rho))

        return numerator / denominator

    def calc_avg_queue_length(self) -> float:
        """计算平均队列长度 Lq"""
        if not self.is_stable():
            return float('inf')

        c = self.num_servers
        rho = self.rho
        Pw = self.calc_prob_wait()

        # Lq = (c*ρ)^c * ρ / (c! * (1-ρ)^2) * Pw
        numerator = ((c * rho) ** c) * rho * Pw
        denominator = self._factorial(c) * ((1 - rho) ** 2)

        return numerator / denominator

    def calc_avg_waiting_time(self) -> float:
        """计算平均等待时间 Wq (小时 -> 秒转换)"""
        Lq = self.calc_avg_queue_length()
        return Lq / self.arrival_rate  # 毫秒

    def calc_avg_response_time(self) -> float:
        """计算平均响应时间 W = Wq + 1/μ"""
        return self.calc_avg_waiting_time() + (1.0 / self.service_rate)

    def calc_throughput(self) -> float:
        """计算吞吐量"""
        return min(self.arrival_rate, self.num_servers * self.service_rate)

    @staticmethod
    def _factorial(n: int) -> float:
        if n <= 1:
            return 1.0
        result = 1.0
        for i in range(2, n + 1):
            result *= i
        return result


class QueueingNetworkAnalyzer:
    """
    排队网络分析器

    用于分析端到端延迟、资源利用率等
    """

    def __init__(self, topology: 'Topology', service_configs: Dict[str, MicroService]):
        self.topology = topology
        self.service_configs = service_configs

    def analyze_latency(self, chain: ServiceChain,
                       deployment: DeploymentPlan) -> Dict[str, float]:
        """
        分析服务链的端到端延迟

        Returns:
            {
                "total_latency": 总延迟 (ms),
                "queuing_delay": 排队延迟 (ms),
                "processing_delay": 处理延迟 (ms),
                "communication_delay": 通信延迟 (ms)
            }
        """
        total_queuing = 0.0
        total_processing = 0.0
        total_communication = 0.0

        for i, service_id in enumerate(chain.services):
            # 获取服务部署的节点
            node_id = self._get_deployed_node(service_id, deployment)
            if not node_id:
                continue

            # 实例数量
            instances = deployment.get_instances(service_id)
            if instances <= 0:
                instances = 1

            service_cfg = self.service_configs[service_id]

            # M/M/C排队延迟
            # 该服务累计到达率 = chain.arrival_rate * (product of routing probabilities)
            # 简化：直接使用chain.arrival_rate
            mmc = MMCQueue(
                arrival_rate=chain.arrival_rate,
                service_rate=service_cfg.processing_rate,
                num_servers=instances
            )

            total_queuing += mmc.calc_avg_waiting_time()
            total_processing += mmc.calc_avg_response_time() - mmc.calc_avg_waiting_time()

            # 通信延迟 (与服务i+1之间)
            if i < len(chain.services) - 1:
                next_service = chain.services[i + 1]
                next_node = self._get_deployed_node(next_service, deployment)
                if next_node and next_node != node_id:
                    comm_delay = self.topology.get_communication_delay(node_id, next_node)
                    total_communication += comm_delay

        return {
            "total_latency": total_queuing + total_processing + total_communication,
            "queuing_delay": total_queuing,
            "processing_delay": total_processing,
            "communication_delay": total_communication
        }

    def analyze_resource_utilization(self, deployment: DeploymentPlan) -> Dict[str, float]:
        """
        分析资源利用率

        Returns:
            {node_id: utilization (0-1)}
        """
        utilizations = {}

        for node_id, node in self.topology.nodes.items():
            used_cpu = deployment.get_node_cpu_usage(node_id, self.service_configs)
            utilization = used_cpu / node.cpu_cores if node.cpu_cores > 0 else 0
            utilizations[node_id] = min(utilization, 1.0)

        return utilizations

    def _get_deployed_node(self, service_id: str, deployment: DeploymentPlan) -> Optional[str]:
        """获取服务部署的节点（返回第一个）"""
        for (s, n), v in deployment.placement.items():
            if s == service_id and v > 0:
                return n
        return None
```

```python
# algorithms/deployment/base.py

class DeploymentAlgorithm(ABC):
    """部署算法基类"""

    def __init__(self, config: Optional[Dict] = None):
        self.config = config or {}

    @abstractmethod
    def deploy(self, topology: Topology,
              service_configs: Dict[str, MicroService],
              chains: List[ServiceChain]) -> DeploymentPlan:
        """
        执行部署

        Args:
            topology: 网络拓扑
            service_configs: 微服务配置 {service_id: MicroService}
            chains: 服务链列表

        Returns:
            DeploymentPlan: 部署方案
        """
        pass

    def validate(self, plan: DeploymentPlan,
                 topology: Topology,
                 service_configs: Dict[str, MicroService]) -> bool:
        """验证部署方案合法性"""
        # 检查节点资源不超限
        for node_id, node in topology.nodes.items():
            used = plan.get_node_cpu_usage(node_id, service_configs)
            if used > node.cpu_cores:
                return False
        return True
```

#### 4.2.1 FFD (First Fit Decreasing)

```python
# algorithms/deployment/baselines/ffd.py

class FirstFitDecreasing(DeploymentAlgorithm):
    """
    First Fit Decreasing (FFD)

    策略:
    1. 按资源需求量降序排列微服务
    2. 对每个微服务，贪心地选择第一个能满足资源需求的节点
    """

    def deploy(self, topology: Topology,
              service_configs: Dict[str, MicroService],
              chains: List[ServiceChain]) -> DeploymentPlan:

        plan = DeploymentPlan()

        # 计算每个服务的总资源需求
        service_demands = {}
        for service_id, cfg in service_configs.items():
            # 简化：统计该服务在所有chain中出现的总到达率作为需求指标
            total_rate = sum(c.arrival_rate for c in chains if service_id in c.services)
            # 所需实例数 = ceil(总到达率 / 处理率)
            instances_needed = max(1, int(total_rate / cfg.processing_rate) + 1)
            service_demands[service_id] = instances_needed * cfg.cpu_per_instance

        # 降序排列
        sorted_services = sorted(service_demands.items(), key=lambda x: -x[1])

        # 分配
        for service_id, cpu_needed in sorted_services:
            cfg = service_configs[service_id]
            instances = service_demands[service_id] // cfg.cpu_per_instance
            if instances < 1:
                instances = 1

            # 贪心选择第一个满足条件的节点
            for node_id, node in topology.nodes.items():
                if node.can_deploy(int(instances * cfg.cpu_per_instance)):
                    plan.add_instance(service_id, node_id, instances)
                    break

        return plan
```

```python
# simulation/engine.py

class SimulationEngine:
    """仿真引擎"""

    def __init__(self, config: Dict):
        self.config = config
        self.topology = None
        self.service_configs = {}
        self.chains = []
        self.deployment_alg = None
        self.routing_alg = None
        self.rl_alg = None
        self.metrics = MetricsCollector()

    def set_topology(self, topology: Topology):
        self.topology = topology

    def set_services(self, service_configs: Dict[str, MicroService]):
        self.service_configs = service_configs

    def set_chains(self, chains: List[ServiceChain]):
        self.chains = chains

    def set_algorithms(self, deployment: str = "ffd",
                       routing: str = "shortest_path",
                       rl: str = None):
        """设置算法"""
        # 加载部署算法
        if deployment == "ffd":
            self.deployment_alg = FirstFitDecreasing()
        elif deployment == "rls":
            self.deployment_alg = RandomLocalSearch()
        elif deployment == "cds":
            self.deployment_alg = CoLocatedDeployment()
        elif deployment == "our":
            self.deployment_alg = OurAlgorithm()  # 待实现
        # ...

        # 加载路由算法
        if routing == "shortest_path":
            self.routing_alg = ShortestPathRouting()
        elif routing == "random":
            self.routing_alg = RandomRouting()
        # ...

    def run(self) -> ExperimentResult:
        """运行实验"""

        # 1. 部署阶段
        deployment_plan = self.deployment_alg.deploy(
            self.topology,
            self.service_configs,
            self.chains
        )

        # 2. 路由阶段
        analyzer = QueueingNetworkAnalyzer(self.topology, self.service_configs)

        results = []
        for chain in self.chains:
            routing_result = self.routing_alg.route(chain, deployment_plan, self.topology)
            latency = analyzer.analyze_latency(chain, deployment_plan)
            results.append({
                "chain_id": chain.chain_id,
                "latency": latency,
                "routing": routing_result
            })

        # 3. 汇总指标
        experiment_result = ExperimentResult()
        experiment_result.deployment_plan = deployment_plan
        experiment_result.routing_results = results
        experiment_result.avg_latency = np.mean([r["latency"]["total_latency"] for r in results])
        experiment_result.resource_utilization = analyzer.analyze_resource_utilization(deployment_plan)

        return experiment_result


@dataclass
class ExperimentResult:
    """实验结果"""
    deployment_plan: DeploymentPlan = None
    routing_results: List[Dict] = field(default_factory=list)
    avg_latency: float = 0.0
    resource_utilization: Dict[str, float] = field(default_factory=dict)
    success_rate: float = 1.0  # 满足延迟约束的比例
    total_cost: float = 0.0    # 部署成本
```
