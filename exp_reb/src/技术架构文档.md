# 服务部署与路由实验框架 技术架构文档

## 1. 概述

本框架用于实现和评测微服务/AI服务的部署算法和路由算法，基于三篇论文的核心方法：

- **论文一** (IEEE TPDS 2023): GMDA部署算法 + RMPR路由算法
- **论文二** (IEEE TSC 2024): DA-MAB任务卸载与模型放置
- **论文三** (GSTC): JCQDA部署算法 + LAMRA动态模型替换

class Node:

# core/service/base.py

class Service:
    """服务基类"""
    def __init__(self, service_id: str, cpu_per_instance: float,
                 gpu_per_instance: float, processing_rate: float):
        self.service_id = service_id
        self.cpu_per_instance = cpu_per_instance   # 每个实例CPU需求
        self.gpu_per_instance = gpu_per_instance   # 每个实例GPU需求
        self.processing_rate = processing_rate      # 处理速率 (请求/秒)
        self.instances = {}                          # node_id -> instance_count

    """服务实例"""
    def __init__(self, service_id: str, node_id: str, instance_id: int):
        self.service_id = service_id
        self.node_id = node_id
        self.instance_id = instance_id
        self.queue_length = 0
        self.arrival_rate = 0.0
        self.service_rate = 0.0

class Request:
    """请求模型"""
    def __init__(self, request_id: str, service_chain: List[str],
                 arrival_rate: float, max_latency: float):
        self.request_id = request_id
        self.service_chain = service_chain    # 服务链 [s1, s2, s3, ...]
        self.arrival_rate = arrival_rate     # 到达率
        self.max_latency = max_latency       # 最大容忍延迟
        self.current_node = None
        self.path = []                       # 路由路径

from abc import ABC, abstractmethod

class DeploymentAlgorithm(ABC):
    """部署算法基类"""

    def __init__(self, config: Dict):
        self.config = config

    @abstractmethod
    def deploy(self, topology: Topology,
              services: List[Service],
              requests: List[Request]) -> DeploymentResult:
        """
        执行部署

        Returns:
            DeploymentResult: 包含部署方案 {service_id: {node_id: instance_count}}
        """
        pass

    def validate_deployment(self, deployment: DeploymentResult,
                          topology: Topology) -> bool:
        """验证部署方案合法性"""
        pass

class DeploymentResult:
    """部署结果"""

    def __init__(self):
        self.placement = {}          # {service_id: {node_id: instance_count}}
        self.total_cost = 0.0
        self.deployment_time = 0.0

    def get_instances(self, service_id: str) -> int:
        """获取服务实例总数"""
        pass

    def get_node_usage(self, node_id: str) -> Dict:
        """获取节点资源使用"""
        pass

## 7. 使用示例

```python
# main.py

from simulation.engine import SimulationEngine
from config.default_config import DEFAULT_CONFIG

def main():
    # 1. 创建仿真引擎
    engine = SimulationEngine(DEFAULT_CONFIG)

    # 2. 加载拓扑（数据中心，20节点）
    engine.load_topology({
        "type": "datacenter",
        "num_nodes": 20,
    })

    # 3. 加载服务（10个微服务）
    engine.load_services({
        "num_services": 10,
    })

    # 4. 设置算法
    engine.set_algorithms(
        deployment="gmda",   # 使用GMDA部署
        routing="rmpr",      # 使用RMPR路由
        dynamic=None         # 静态实验
    )

    # 5. 运行实验
    result = engine.run_static_experiment()

    # 6. 打印结果
    print(f"平均延迟: {result.avg_latency:.2f} ms")
    print(f"成功率: {result.success_rate:.2%}")
    print(f"资源利用率: {result.resource_utilization}")

    # 7. 保存结果
    result.save("results/gmda_rmpr_baseline.json")

if __name__ == "__main__":
    main()
```
