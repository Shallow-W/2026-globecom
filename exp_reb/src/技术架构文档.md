# 服务部署与路由实验框架 技术架构文档

## 1. 概述

本框架用于实现和评测微服务系统的部署算法，通过**解析计算**方式评估算法性能。

**核心思路**: 部署算法输出部署方案 → 排队模型直接计算各项指标

```
拓扑配置 + 服务配置 + 服务链配置
         ↓
   部署算法 (DeploymentAlgorithm)
         ↓
    DeploymentPlan {service_id → {node_id → count}}
         ↓
   排队网络分析器 (QueueingNetworkAnalyzer)
         ↓
    解析计算: M/M/C延迟 + 通信延迟
         ↓
   指标汇总: 平均延迟、资源利用率、成功率、部署成本
```

```python
# core/topology/node.py

class Node:
    """计算节点（支持CPU + GPU异构资源）"""
    def __init__(self, node_id: str, cpu_cores: int, gpu_memory: int = 0):
        self.node_id = node_id
        self.cpu_cores = cpu_cores          # CPU核心数
        self.gpu_memory = gpu_memory        # GPU内存 (MB)
        self.used_cpu = 0
        self.used_gpu = 0
        self.services = {}                  # {service_id: instance_count}

    @property
    def available_cpu(self) -> int:
        return self.cpu_cores - self.used_cpu

    @property
    def available_gpu(self) -> int:
        return self.gpu_memory - self.used_gpu

    def can_deploy(self, cpu_needed: int, gpu_needed: int = 0) -> bool:
        """检查是否可以部署（同时满足CPU和GPU）"""
        return self.available_cpu >= cpu_needed and self.available_gpu >= gpu_needed

    def deploy(self, service_id: str, instances: int,
               cpu_per_instance: float, gpu_per_instance: int = 0):
        self.services[service_id] = instances
        self.used_cpu += int(instances * cpu_per_instance)
        self.used_gpu += instances * gpu_per_instance

    def remove_service(self, service_id: str):
        if service_id in self.services:
            del self.services[service_id]
```

```python
# core/topology/topology.py

class Topology:
    """网络拓扑"""
    def __init__(self):
        self.nodes: Dict[str, Node] = {}
        self.links: List[Link] = []
        self.adjacency: Dict[str, List[str]] = {}  # 邻接表
        self.link_map: Dict[Tuple[str, str], Link] = {}  # 快速查找链路

    def add_node(self, node: Node):
        self.nodes[node.node_id] = node
        self.adjacency[node.node_id] = []

    def add_link(self, link: Link):
        self.links.append(link)
        self.adjacency[link.src].append(link.dst)
        self.link_map[(link.src, link.dst)] = link

    def get_shortest_path(self, src: str, dst: str) -> List[str]:
        """Dijkstra最短路径"""
        import heapq

        distances = {n: float('inf') for n in self.nodes}
        distances[src] = 0
        previous = {n: None for n in self.nodes}
        pq = [(0, src)]

        while pq:
            d, u = heapq.heappop(pq)
            if d > distances[u]:
                continue
            if u == dst:
                break
            for v in self.adjacency.get(u, []):
                link = self.link_map.get((u, v))
                if link:
                    nd = d + link.delay
                    if nd < distances[v]:
                        distances[v] = nd
                        previous[v] = u
                        heapq.heappush(pq, (nd, v))

        # 重建路径
        if previous[dst] is None and dst != src:
            return []
        path = []
        cur = dst
        while cur:
            path.append(cur)
            cur = previous[cur]
        path.reverse()
        return path

    def get_communication_delay(self, src: str, dst: str) -> float:
        """获取两节点间通信延迟"""
        if src == dst:
            return 0.0
        path = self.get_shortest_path(src, dst)
        if not path or len(path) < 2:
            return 0.0
        return sum(self.link_map[(path[i], path[i+1])].delay for i in range(len(path)-1))

    def get_nodes_by_resource(self, cpu_needed: int) -> List[Node]:
        return [n for n in self.nodes.values() if n.can_deploy(cpu_needed)]
```

```python
# core/service/deployment.py

@dataclass
class DeploymentPlan:
    """
    部署方案 (包含模型版本信息)

    Our算法: 动态选择模型版本
    Baseline算法: 固定使用Model-M
    """
    # 部署映射: {(service_id, node_id): {version_id: count}}
    placement: Dict[Tuple[str, str], Dict[str, int]] = field(default_factory=dict)

    def add(self, service_id: str, node_id: str, version_id: str, count: int = 1):
        """添加部署实例"""
        key = (service_id, node_id)
        if key not in self.placement:
            self.placement[key] = {}
        self.placement[key][version_id] = self.placement[key].get(version_id, 0) + count

    def get_service_instances(self, service_id: str, version_id: str = None) -> int:
        """获取服务总实例数 (可指定版本)"""
        if version_id:
            return sum(v.get(version_id, 0) for (s, _), v in self.placement.items() if s == service_id)
        return sum(sum(v.values()) for (s, _), v in self.placement.items() if s == service_id)

    def get_node_instances(self, node_id: str) -> Dict[str, Dict[str, int]]:
        """获取节点上部署的服务及版本"""
        return {s: v for (s, n), v in self.placement.items() if n == node_id}

    def get_version_usage(self) -> Dict[str, int]:
        """统计各版本使用量"""
        usage = {}
        for (s, n), versions in self.placement.items():
            for vid, count in versions.items():
                usage[vid] = usage.get(vid, 0) + count
        return usage
        """节点上部署的服务"""
        return {s: v for (s, n), v in self.placement.items() if n == node_id}

    def get_node_cpu_usage(self, node_id: str, services: Dict[str, MicroService]) -> int:
        """节点CPU使用量"""
        total = 0
        for (s, n), v in self.placement.items():
            if n == node_id and s in services:
                total += int(v * services[s].cpu_per_instance)
        return total

    def get_node_gpu_usage(self, node_id: str, services: Dict[str, MicroService]) -> int:
        """节点GPU使用量 (MB)"""
        total = 0
        for (s, n), v in self.placement.items():
            if n == node_id and s in services:
                total += v * services[s].gpu_per_instance
        return total

    def validate(self, topology: 'Topology', services: Dict[str, MicroService]) -> bool:
        """验证部署合法性（CPU + GPU）"""
        for node_id, node in topology.nodes.items():
            cpu_used = self.get_node_cpu_usage(node_id, services)
            gpu_used = self.get_node_gpu_usage(node_id, services)
            if cpu_used > node.cpu_cores or gpu_used > node.gpu_memory:
                return False
        return True
```

### 3.3 排队模型 (core/queueing)

```python
# core/queueing/mmc.py

class MMCQueue:
    """
    M/M/C 排队模型 (无限容量)

    输入: 到达率λ, 服务率μ, 服务器数c
    输出: 等待概率、平均队列长度、平均等待时间、平均响应时间
    """

    def __init__(self, arrival_rate: float, service_rate: float, num_servers: int):
        self.lambda_ = arrival_rate   # λ
        self.mu = service_rate        # μ (每个服务器)
        self.c = num_servers          # c

    @property
    def rho(self) -> float:
        """服务强度 ρ = λ / (c * μ)"""
        return self.lambda_ / (self.c * self.mu)

    def is_stable(self) -> bool:
        return self.rho < 1.0

    def erlang_c(self) -> float:
        """
        Erlang-C 公式: 等待概率 Pw
        Pw = ((cρ)^c / c!) / (Σ_{k=0}^{c-1} ((cρ)^k / k!) + ((cρ)^c / c!) / (1-ρ))
        """
        if self.rho >= 1.0:
            return 1.0
        c = self.c
        rho = self.rho

        # 分子
        num = (c * rho) ** c / self._factorial(c)

        # 分母前半部分 (k=0 到 c-1)
        sum_part = sum((c * rho) ** k / self._factorial(k) for k in range(c))

        # 分母后半部分
        denom = sum_part + (c * rho) ** c / (self._factorial(c) * (1 - rho))

        return num / denom

    def avg_queue_length(self) -> float:
        """平均队列长度 Lq = (cρ)^c * ρ * Pw / (c! * (1-ρ)^2)"""
        if not self.is_stable():
            return float('inf')
        c = self.c
        rho = self.rho
        pw = self.erlang_c()
        return ((c * rho) ** c * rho * pw) / (self._factorial(c) * (1 - rho) ** 2)

    def avg_waiting_time(self) -> float:
        """平均等待时间 Wq = Lq / λ (ms)"""
        if self.lambda_ <= 0:
            return 0.0
        return (self.avg_queue_length() / self.lambda_) * 1000  # 转换为ms

    def avg_response_time(self) -> float:
        """平均响应时间 W = Wq + 1/μ (ms)"""
        return self.avg_waiting_time() + (1000.0 / self.mu)

    def calc(self) -> Dict[str, float]:
        """计算所有指标"""
        return {
            "rho": self.rho,
            "pw": self.erlang_c(),
            "Lq": self.avg_queue_length(),
            "Wq": self.avg_waiting_time(),
            "W": self.avg_response_time(),
            "stable": self.is_stable()
        }

    @staticmethod
    def _factorial(n: int) -> float:
        if n <= 1:
            return 1.0
        result = 1.0
        for i in range(2, n + 1):
            result *= i
        return result
```

```python
# core/queueing/analyzer.py

class QueueingNetworkAnalyzer:
    """
    排队网络分析器

    给定部署方案，计算服务链的端到端延迟等指标
    """

    def __init__(self, topology: Topology, services: Dict[str, MicroService]):
        self.topology = topology
        self.services = services

    def calc_chain_latency(self, chain: ServiceChain,
                          deployment: DeploymentPlan) -> Dict[str, float]:
        """
        计算服务链的端到端延迟

        延迟组成:
        - 排队延迟: M/M/C 模型的 Wq
        - 处理延迟: Σ (1/μ_i)
        - 通信延迟: 服务间传输延迟

        Returns:
            {
                "total": 总延迟 (ms),
                "queuing": 排队延迟 (ms),
                "processing": 处理延迟 (ms),
                "communication": 通信延迟 (ms)
            }
        """
        total_queuing = 0.0
        total_processing = 0.0
        total_comm = 0.0

        for i, service_id in enumerate(chain.services):
            # 获取服务部署的节点
            node_id = self._find_service_node(service_id, deployment)
            if not node_id:
                continue

            service_cfg = self.services[service_id]
            instances = deployment.get_service_instances(service_id)
            if instances <= 0:
                instances = 1

            # M/M/C 排队延迟
            mmc = MMCQueue(
                arrival_rate=chain.arrival_rate,
                service_rate=service_cfg.processing_rate,
                num_servers=instances
            )
            calc = mmc.calc()

            total_queuing += calc["Wq"]
            total_processing += calc["W"] - calc["Wq"]  # 1/μ

            # 通信延迟 (服务i到服务i+1)
            if i < chain.length - 1:
                next_service = chain.services[i + 1]
                next_node = self._find_service_node(next_service, deployment)
                if next_node and next_node != node_id:
                    total_comm += self.topology.get_communication_delay(node_id, next_node)

        return {
            "total": total_queuing + total_processing + total_comm,
            "queuing": total_queuing,
            "processing": total_processing,
            "communication": total_comm
        }

    def calc_resource_utilization(self, deployment: DeploymentPlan) -> Dict[str, float]:
        """计算各节点资源利用率"""
        utils = {}
        for node_id, node in self.topology.nodes.items():
            used = deployment.get_node_cpu_usage(node_id, self.services)
            utils[node_id] = used / node.cpu_cores if node.cpu_cores > 0 else 0
        return utils

    def _find_service_node(self, service_id: str, deployment: DeploymentPlan) -> Optional[str]:
        """查找服务部署的节点"""
        for (s, n), v in deployment.placement.items():
            if s == service_id and v > 0:
                return n
        return None
```

```python
# algorithms/deployment/ours.py

class ModelSearcher:
    """
    模型搜索器 - Our算法的核心组件

    根据Excel模型库和当前约束，搜索满足条件的候选模型
    """

    def __init__(self, excel_path: str):
        self.excel_path = excel_path
        self.model_tables = {}  # {task_type: DataFrame}

    def load_models(self, task_type: str):
        """从Excel加载指定任务类型的模型库"""
        import pandas as pd
        self.model_tables[task_type] = pd.read_excel(
            self.excel_path, sheet_name=task_type
        )

    def calc_F_max(self, C_max: float, lambda_t: float, T_SLA: float) -> float:
        """
        动态算力红线 (F_max)

        F_max(t) = C_max / (λ(t) + 1/T_SLA)

        流量λ越大 → 算力红线越收缩 → 只能选轻量模型
        """
        return C_max / (lambda_t + 1/T_SLA)

    def calc_w2(self, lambda_t: float, lambda_th: float = 20.0, alpha: float = 1.0) -> float:
        """动态流量惩罚权重 w2(t) = α·exp(max(0, λ-λ_th)/λ_th)"""
        return alpha * math.exp(max(0, lambda_t - lambda_th) / lambda_th)

    def calc_w3(self, M_used: float, M_total: float, beta: float = 1.0) -> float:
        """动态内存惩罚权重 w3(t) = β·(M_used/M_total)"""
        return beta * (M_used / M_total) if M_total > 0 else 0

    def search(self, task_type: str, M_max: float, C_max: float,
               lambda_t: float, T_SLA: float, M_used: float = 0, M_total: float = 1) -> List[Dict]:
        """
        根据当前节点状态搜索候选模型

        Args:
            task_type: 任务类型 (class_scene等)
            M_max: 节点可用内存
            C_max: 节点峰值算力
            lambda_t: 当前到达率
            T_SLA: SLA延迟约束
            M_used: 已使用内存
            M_total: 总内存

        Returns:
            List[Dict]: 满足约束的候选模型列表，按效用降序排列
        """
        import pandas as pd

        # Step 1: 计算动态算力红线
        F_max = self.calc_F_max(C_max, lambda_t, T_SLA)

        # Step 2: 从Excel获取候选模型
        if task_type not in self.model_tables:
            self.load_models(task_type)
        candidates = self.model_tables[task_type]

        # Step 3: 硬约束过滤 (内存 + 算力)
        # Excel列: architecture=model_id, proxy_score=S_proxy, model_params=P_model, flops=F_flops
        filtered = candidates[
            (candidates['model_params'] <= M_max) &   # 内存约束
            (candidates['flops'] <= F_max)              # 算力约束
        ]

        if filtered.empty:
            return []

        # Step 4: 计算动态权重
        w2 = self.calc_w2(lambda_t)
        w3 = self.calc_w3(M_used, M_total)
        w1 = 1.0 / (1.0 + w2 + w3)

        # Step 5: 计算效用函数并排序
        # U(a_k) = w1·S_proxy - w2·F_flops - w3·P_model
        def calc_utility(row):
            return (w1 * row['proxy_score']
                    - w2 * row['flops']
                    - w3 * row['model_params'])

        filtered = filtered.copy()
        filtered['utility'] = filtered.apply(calc_utility, axis=1)
        filtered = filtered.sort_values('utility', ascending=False)

        return filtered.to_dict('records')


class OurDeployment:
    """
    Our部署算法 - 根据globecom.pdf的启发式调度

    三阶段流水线:
    1. 硬约束过滤 (内存红线 + 算力红线)
    2. 动态启发式加权选择 (效用函数最大化)
    3. 输出部署方案
    """

    def __init__(self, model_searcher: ModelSearcher):
        self.model_searcher = model_searcher

    def deploy(self, topology: 'Topology',
               services: Dict[str, 'Service'],
               chains: List['ServiceChain']) -> DeploymentPlan:
        """
        执行Our部署算法

        Args:
            topology: 网络拓扑
            services: 服务配置 {service_id: Service}
            chains: 服务链列表

        Returns:
            DeploymentPlan: 部署方案
        """
        deployment_plan = DeploymentPlan()

        # 预加载所有任务类型的模型库
        task_types = set()
        for chain in chains:
            # 从chain获取task_type（每条链关联一个任务类型）
            if hasattr(chain, 'task_type'):
                task_types.add(chain.task_type)
        for task_type in task_types:
            self.model_searcher.load_models(task_type)

        # 为每个节点进行部署决策
        for node_id, node in topology.nodes.items():
            M_max = node.available_memory
            C_max = node.cpu_cores  # 或使用GPU算力
            M_used = node.used_memory if hasattr(node, 'used_memory') else 0
            M_total = node.gpu_memory if hasattr(node, 'gpu_memory') else M_max

            for chain in chains:
                # 获取链的任务类型
                task_type = getattr(chain, 'task_type', 'default')
                if not task_type or task_type not in self.model_searcher.model_tables:
                    continue

                # 搜索候选模型
                candidates = self.model_searcher.search(
                    task_type=task_type,
                    M_max=M_max,
                    C_max=C_max,
                    lambda_t=chain.arrival_rate,
                    T_SLA=chain.max_latency,
                    M_used=M_used,
                    M_total=M_total
                )

                if candidates:
                    # 选择效用最大的模型
                    best_model = candidates[0]
                    deployment_plan.add(
                        service_id=chain.chain_id,  # 或具体service_id
                        node_id=node_id,
                        version_id=best_model['architecture'],
                        count=1
                    )

        return deployment_plan


class OurRouting:
    """
    Our路由算法 - 基于负载感知的路由

    策略:
    - 优先选择当前负载较低的节点
    - 考虑路径延迟
    - 避免过载
    """

    def route(self, chain: 'ServiceChain',
              deployment_plan: DeploymentPlan,
              topology: 'Topology') -> Dict[str, List[Tuple[str, float]]]:
        """
        为服务链确定路由路径

        Returns:
            Dict[str, List[Tuple[node_id, probability]]]: 各服务的路由概率
        """
        routing = {}

        for service_id in chain.services:
            # 获取部署了该服务的所有节点
            candidate_nodes = []
            for (s, n), versions in deployment_plan.placement.items():
                if s == service_id and versions:
                    candidate_nodes.append(n)

            if not candidate_nodes:
                continue

            # 计算各节点的路由权重
            weights = []
            for node_id in candidate_nodes:
                # 负载越低、路径越短，权重越高
                node = topology.nodes.get(node_id)
                if not node:
                    continue

                # 获取节点负载率
                load = getattr(node, 'load', 0.5)
                # 获取到该节点的路径延迟
                path_delay = topology.get_communication_delay(
                    chain.entry_node, node_id
                ) if hasattr(chain, 'entry_node') else 0

                # 权重计算: 1 / (load * path_delay + ε)
                weight = 1.0 / (load * path_delay + 1e-6)
                weights.append((node_id, weight))

            # 归一化权重
            total = sum(w for _, w in weights)
            if total > 0:
                routing[service_id] = [(n, w/total) for n, w in weights]

        return routing


class OurAlgorithm(DeploymentAlgorithm):
    """
    Our完整算法 (部署 + 路由) - 基于globecom.pdf

    对应globecom.pdf的云边协同按需调度框架:
    - 模型搜索 ≈ 零成本代理评估 (从Excel搜索)
    - Deployment ≈ 启发式加权选择部署
    - Routing ≈ 请求路由
    """

    def __init__(self, config: Optional[Dict] = None):
        super().__init__(config)
        self.fixed_version = None  # Our算法不固定版本

        # 初始化模型搜索器
        excel_path = config.get('excel_model_path') if config else None
        if excel_path:
            self.model_searcher = ModelSearcher(excel_path)
        else:
            self.model_searcher = None

        self.deployment = None
        self.routing = OurRouting()

    def deploy(self, topology: Topology,
              services: Dict[str, Service],
              chains: List[ServiceChain]) -> DeploymentPlan:
        """
        执行Our完整算法

        流程:
        1. 加载模型库
        2. 部署 (硬约束过滤 + 动态启发式)
        3. 路由 (负载感知)
        """
        # Step 1: 初始化部署器
        if not self.deployment and self.model_searcher:
            self.deployment = OurDeployment(self.model_searcher)

        if not self.deployment:
            raise ValueError("ModelSearcher not initialized. Set excel_model_path in config.")

        # Step 2: 执行部署
        deployment_plan = self.deployment.deploy(topology, services, chains)

        return deployment_plan

    def route(self, chain: ServiceChain,
              deployment_plan: DeploymentPlan,
              topology: Topology) -> Dict[str, List[Tuple[str, float]]]:
        """为服务链计算路由方案"""
        return self.routing.route(chain, deployment_plan, topology)

    def solve(self, topology: Topology,
             services: Dict[str, Service],
             chains: List[ServiceChain]) -> Dict[str, Any]:
        """
        完整求解: 部署 + 路由

        Returns:
            Dict: 包含deployment_plan和routing_plan
        """
        # 部署
        deployment_plan = self.deploy(topology, services, chains)

        # 路由
        routing_plan = {}
        for chain in chains:
            routing_plan[chain.chain_id] = self.route(
                chain, deployment_plan, topology
            )

        return {
            'deployment_plan': deployment_plan,
            'routing_plan': routing_plan
        }
```

#### 关键公式

| 公式 | 名称 | 说明 |
|------|------|------|
| F_max(t) = C_max / (λ(t) + 1/T_SLA) | 动态算力红线 | 流量越大，可选模型越轻 |
| U(a_k) = w1·S_proxy - w2·F_flops - w3·P_model | 效用函数 | 精度↑、计算量↓、参数量↓则优 |
| w2(t) = α·exp(max(0, λ-λ_th)/λ_th) | 流量惩罚权重 | 高流量时惩罚计算量 |
| w3(t) = β·(M_used/M_total) | 内存惩罚权重 | 内存紧张时惩罚参数量 |
| w1(t) = 1/(1+w2+w3) | 归一化精度权重 | 保证权重和为1 |
| μ = C_max / F_flops | FLOPs→服务率 | 算力换算服务率 |

```python
# algorithms/deployment/baselines/ffd_m.py

class FirstFitDecreasing(DeploymentAlgorithm):
    """
    First Fit Decreasing (FFD-M)

    Baseline算法: 固定使用 Model-M

    策略:
    1. 计算每个微服务需要的资源量 (使用 Model-M 的参数)
    2. 按资源量降序排列
    3. 对每个服务，贪心选择第一个能满足资源需求的节点
    """

    def deploy(self, topology: Topology,
              services: Dict[str, Service],
              chains: List[ServiceChain]) -> DeploymentPlan:

        plan = DeploymentPlan()
        fixed_version = "Model-M"  # 固定使用中精度模型

        # Step 1: 计算每个服务的资源需求 (使用Model-M版本)
        service_demand = {}
        for service_id, cfg in services.items():
            version = cfg.get_version(fixed_version)

            # 累计到达率
            total_rate = sum(c.arrival_rate for c in chains if service_id in c.services)
            # 所需实例数
            instances = max(1, int(total_rate / version.mu) + 1)
            cpu_needed = int(instances * version.cpu_per_instance)
            gpu_needed = instances * version.gpu_per_instance

            service_demand[service_id] = {
                "version": fixed_version,
                "instances": instances,
                "cpu": cpu_needed,
                "gpu": gpu_needed
            }
            gpu_needed = instances * cfg.gpu_per_instance  # GPU内存需求
            service_demand[service_id] = {
                "instances": instances,
                "cpu": cpu_needed,
                "gpu": gpu_needed
            }

        # Step 2: 降序排列（按总资源量）
        sorted_services = sorted(
            service_demand.items(),
            key=lambda x: -(x[1]["cpu"] + x[1]["gpu"])
        )

        # Step 3: 贪心部署
        for service_id, demand in sorted_services:
            cfg = services[service_id]
            for node_id, node in topology.nodes.items():
                if node.can_deploy(demand["cpu"], demand["gpu"]):
                    plan.add(service_id, node_id, demand["instances"])
                    # 更新节点状态
                    node.deploy(service_id, demand["instances"],
                               cfg.cpu_per_instance, cfg.gpu_per_instance)
                    break

        return plan
```

```python
class CoLocatedDeployment(DeploymentAlgorithm):
    """
    Co-Located Deployment (CDS)

    策略:
    - 将同一服务链的所有服务尽可能部署在同一节点上
    - 减少跨节点通信延迟
    """

    def deploy(self, topology: Topology,
              services: Dict[str, MicroService],
              chains: List[ServiceChain]) -> DeploymentPlan:

        plan = DeploymentPlan()

        # 按链分组服务
        chain_services = {}  # node_id -> {service_id: count}
        for chain in chains:
            # 找到一条链经过的所有服务
            services_in_chain = set(chain.services)

            # 找一个有足够资源的节点部署这些服务
            best_node = self._find_best_node(topology, services, services_in_chain)
            if best_node:
                for service_id in services_in_chain:
                    cfg = services[service_id]
                    instances = max(1, int(chain.arrival_rate / cfg.processing_rate))
                    plan.add(service_id, best_node, instances)
                    best_node.deploy(service_id, instances, cfg.cpu_per_instance)

        return plan

    def _find_best_node(self, topology, services, needed_services) -> Optional[str]:
        """找能容纳这些服务的最佳节点"""
        total_cpu = sum(services[s].cpu_per_instance for s in needed_services)
        # 选择通信开销最小且资源足够的节点
        # 简化：选第一个满足条件的节点
        for node_id, node in topology.nodes.items():
            if node.can_deploy(int(total_cpu)):
                return node_id
        return None
```

```python
# experiments/runner.py

class ExperimentRunner:
    """
    实验运行器

    流程:
    1. 加载配置
    2. 生成拓扑和服务
    3. 运行各种部署算法
    4. 用排队模型计算指标
    5. 汇总结果
    """

    def __init__(self, config: Dict):
        self.config = config
        self.results = []

    def run_single(self, deployment_name: str,
                   topology: Topology,
                   services: Dict[str, MicroService],
                   chains: List[ServiceChain]) -> Dict:
        """运行单个实验"""

        # 获取部署算法
        alg = self._create_deployment_algorithm(deployment_name)

        # 执行部署
        plan = alg.deploy(topology, services, chains)

        # 排队分析
        analyzer = QueueingNetworkAnalyzer(topology, services)

        # 计算每条链的延迟
        chain_latencies = []
        for chain in chains:
            lat = analyzer.calc_chain_latency(chain, plan)
            chain_latencies.append({
                "chain_id": chain.chain_id,
                **lat
            })

        # 汇总指标
        total_lat = sum(c["total"] for c in chain_latencies)
        avg_lat = total_lat / len(chain_latencies) if chain_latencies else 0

        # 资源利用率
        util = analyzer.calc_resource_utilization(plan)

        # 成功率 (满足延迟约束)
        success = sum(1 for c in chain_latencies if c["total"] <=
                     next(ch.max_latency for ch in chains if ch.chain_id == c["chain_id"]))
        success_rate = success / len(chains) if chains else 0

        # 部署成本 (使用节点数)
        cost = len(set(n for (_, n) in plan.placement.keys()))

        return {
            "algorithm": deployment_name,
            "deployment_plan": plan,
            "chain_latencies": chain_latencies,
            "avg_latency": avg_lat,
            "resource_utilization": util,
            "success_rate": success_rate,
            "deployment_cost": cost
        }

    def run_comparison(self,
                      topology: Topology,
                      services: Dict[str, MicroService],
                      chains: List[ServiceChain],
                      algorithms: List[str]) -> List[Dict]:
        """运行多算法对比实验"""
        results = []
        for alg_name in algorithms:
            result = self.run_single(alg_name, topology, services, chains)
            results.append(result)
        return results

    def run_perturbation(self,
                        base_config: Dict,
                        param_name: str,
                        param_values: List,
                        algorithms: List[str]) -> List[Dict]:
        """运行扰动实验"""
        results = []
        for value in param_values:
            # 修改配置
            config = base_config.copy()
            config[param_name] = value

            # 生成数据
            topology = self._generate_topology(config)
            services = self._generate_services(config)
            chains = self._generate_chains(config)

            # 运行实验
            exp_results = self.run_comparison(topology, services, chains, algorithms)
            for r in exp_results:
                r["param"] = param_name
                r["value"] = value
            results.extend(exp_results)

        return results

    def _create_deployment_algorithm(self, name: str) -> DeploymentAlgorithm:
        if name == "ffd":
            return FirstFitDecreasing()
        elif name == "cds":
            return CoLocatedDeployment()
        elif name == "random":
            return RandomDeployment()
        elif name == "greedy":
            return SimpleGreedy()
        elif name == "our":
            return OurAlgorithm()
        # ... 其他算法
        else:
            raise ValueError(f"Unknown algorithm: {name}")
```

```python
# experiments/generator.py

class DataGenerator:
    """测试数据生成器"""

    def __init__(self, seed: int = 42):
        self.rng = np.random.RandomState(seed)

    def generate_topology(self, config: Dict) -> Topology:
        """生成网络拓扑"""
        num_nodes = config.get("num_nodes", 20)
        cpu_range = config.get("cpu_cores_range", [8, 64])
        delay_range = config.get("link_delay_range", [0.1, 5.0])

        topo = Topology()

        # 添加节点
        for i in range(num_nodes):
            cpu = self.rng.randint(cpu_range[0], cpu_range[1] + 1)
            topo.add_node(Node(f"n{i}", cpu))

        # 添加链路 (全连接简化)
        for i in range(num_nodes):
            for j in range(i + 1, num_nodes):
                delay = self.rng.uniform(delay_range[0], delay_range[1])
                topo.add_link(Link(f"n{i}", f"n{j}", delay))
                topo.add_link(Link(f"n{j}", f"n{i}", delay))

        return topo

    def generate_services(self, config: Dict) -> Dict[str, MicroService]:
        """生成微服务配置"""
        num_services = config.get("num_services", 10)
        cpu_range = config.get("cpu_per_instance_range", [1, 4])
        rate_range = config.get("processing_rate_range", [5, 20])

        services = {}
        for i in range(num_services):
            cpu = self.rng.uniform(cpu_range[0], cpu_range[1])
            rate = self.rng.uniform(rate_range[0], rate_range[1])
            services[f"s{i}"] = MicroService(
                service_id=f"s{i}",
                cpu_per_instance=cpu,
                processing_rate=rate
            )
        return services

    def generate_chains(self, config: Dict,
                        services: Dict[str, MicroService]) -> List[ServiceChain]:
        """生成服务链"""
        num_chains = config.get("num_chains", 5)
        rate_range = config.get("arrival_rate_range", [1, 50])
        length_range = config.get("chain_length_range", [2, 5])
        max_latency = config.get("max_latency", 100.0)

        chains = []
        service_ids = list(services.keys())

        for i in range(num_chains):
            length = self.rng.randint(length_range[0], length_range[1] + 1)
            length = min(length, len(service_ids))

            # 随机选择服务组成链
            chain_services = self.rng.choice(service_ids, length, replace=False).tolist()

            rate = self.rng.uniform(rate_range[0], rate_range[1])

            chains.append(ServiceChain(
                chain_id=f"c{i}",
                services=chain_services,
                arrival_rate=rate,
                max_latency=max_latency
            ))

        return chains
```

```python
                alg_metrics[alg] = []

                "mean": np.mean(values),
                "std": np.std(values),
                "min": np.min(values),
                "max": np.max(values)
            }
```

```python
    @staticmethod
    def plot_algorithm_comparison(results: List[Dict], output_dir: str):
        """绘制算法对比图"""
        df = Evaluator.summarize(results)

        fig, axes = plt.subplots(1, 3, figsize=(15, 5))

        # 延迟对比
        df.plot.bar(x="algorithm", y="avg_latency", ax=axes[0], color="steelblue")
        axes[0].set_title("Average Latency")
        axes[0].set_ylabel("Latency (ms)")

        # 成功率对比
        df.plot.bar(x="algorithm", y="success_rate", ax=axes[1], color="green")
        axes[1].set_title("Success Rate")
        axes[1].set_ylabel("Rate")

        # 资源利用率对比
        df.plot.bar(x="algorithm", y="avg_utilization", ax=axes[2], color="orange")
        axes[2].set_title("Resource Utilization")
        axes[2].set_ylabel("Utilization")

        plt.tight_layout()
        plt.savefig(f"{output_dir}/algorithm_comparison.png")
        plt.close()

    @staticmethod
    def plot_perturbation(results: List[Dict], output_dir: str):
        """绘制扰动实验结果"""
        df = pd.DataFrame(results)
        pivot = df.pivot_table(values="avg_latency",
                              index="value",
                              columns="algorithm")
        pivot.plot()
        plt.xlabel(df["param"].iloc[0])
        plt.ylabel("Average Latency (ms)")
        plt.savefig(f"{output_dir}/perturbation.png")
        plt.close()
```
